	//this makes the shell drop down
	private void schelp() {
		add(schelp, 0, -900);
		while (schelp.getY() < -485) {
			schelp.move(0, 4);
			pause(8);
			wolken();
		}
		for (int i = 0; i < 40; i++) {
			schelp.move(0.35, 2);
			pause(8);
			wolken();
		}
		while (schelp.getY() < 0) {
			for (int i = 0; i < 52; i++) {
				schelp.move(-0.35, 2);
				pause(8);
				wolken();
			}
			for (int i = 0; i < 52; i++) {
				schelp.move(0.35, 2);
				pause(8);
				wolken();
			}
		}
	}

	//this makes the chest drop down
	private void chest() {

	//this will make the fish chase the shark
	private void chase() {
		add(vis, 1000, 500);
		add(shark, 1100, 400);
		while (vis.getX() > -500) {
			for (int i = 0; i < 150; i++) {
				vis.move(-1, 0.30);
				shark.move(-1, 0.30);
				pause(4);
				wolken();
			}
			for (int i = 0; i < 50; i++) {
				vis.move(-1, -0.30);
				shark.move(-1, 0.30);
				pause(4);
				wolken();
			}
			for (int i = 0; i < 150; i++) {
				vis.move(-1, -0.30);
				shark.move(-1, -0.30);
				pause(4);
				wolken();
			}
			for (int i = 0; i < 50; i++) {
				vis.move(-1, 0.30);
				shark.move(-1, -0.30);
				pause(4);
				wolken();
			}
		}
	}
